            DatasetItem(0, image=np.ones([2, 2, 3]) * 0),
            DatasetItem(1, image=np.ones([2, 2, 3]) * 1),
        ], categories=['a', 'b'])
        source_dataset.save(source_url, save_images=True)

        project.import_source('source', source_url, format=DEFAULT_FORMAT)
